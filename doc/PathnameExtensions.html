def search(query = nil, predicate = nil)
  matches = []
  query   = Regexp.new(query) if query
  find {|path|
    matched = !query || query.match(path.to_s)
    matched = false if path == self
    matches << path if predicate && predicate.call(path) || matched
  }
  matches
end
